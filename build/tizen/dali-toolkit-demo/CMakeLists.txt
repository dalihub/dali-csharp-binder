SET(TOOLKIT_DEMO_TARGET dali2-csharp-binder-toolkit-demo)

SET(TOOLKIT_DEMO_SOURCES ${dali_csharp_binder_toolkit_demo_src_files})

ADD_LIBRARY(${TOOLKIT_DEMO_TARGET} SHARED ${TOOLKIT_DEMO_SOURCES} )

TARGET_LINK_LIBRARIES(${TOOLKIT_DEMO_TARGET} PUBLIC
  ${DLOG_LDFLAGS}
  ${DALICORE_LDFLAGS}
  ${DALIADAPTOR_LDFLAGS}
  ${DALITOOLKIT_LDFLAGS}
  ${DALITOOLKITDEMO_LDFLAGS})

  ${DLOG_CFLAGS}
  ${DALICORE_CFLAGS}
  ${DALIADAPTOR_CFLAGS}
  ${DALITOOLKIT_CFLAGS}
  ${DALI_PROFILE_CFLAGS}
  ${DALITOOLKITDEMO_CFLAGS}
  ${}
  -I../../../
  -Wall
)

INSTALL(TARGETS ${TOOLKIT_DEMO_TARGET} DESTINATION ${LIB_DIR} )
