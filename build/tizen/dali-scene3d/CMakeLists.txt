SET(SCENE3D_TARGET dali2-csharp-binder-scene3d)

SET( SCENE3D_SOURCES ${dali_csharp_binder_scene3d_src_files})

ADD_LIBRARY( ${SCENE3D_TARGET} SHARED ${SCENE3D_SOURCES} )

# Add Dependency with dali2-csharp-binder.
IF( ENABLE_PKG_CONFIGURE )
  PKG_CHECK_MODULES(DALICSHARPBINDER REQUIRED ${name})

  ${DLOG_LDFLAGS}
  ${DALICORE_LDFLAGS}
  ${DALIADAPTOR_LDFLAGS}
  ${DALITOOLKIT_LDFLAGS}
  ${DALISCENE3D_LDFLAGS}
  ${DALICSHARPBINDER_LDFLAGS}
)

TARGET_COMPILE_OPTIONS( ${SCENE3D_TARGET} PUBLIC
  ${DLOG_CFLAGS}
  ${DALICORE_CFLAGS}
  ${DALIADAPTOR_CFLAGS}
  ${DALITOOLKIT_CFLAGS}
  ${DALISCENE3D_CFLAGS}
  ${DALICSHARPBINDER_CFLAGS}
  ${DALI_PROFILE_CFLAGS}
  ${SCENE3D_CFLAGS}
  ${}
  -I../../../
  -Wall
)

INSTALL( TARGETS ${SCENE3D_TARGET} DESTINATION ${LIB_DIR} )
